```typescript
async getAnalytics(): Promise<AdminAnalytics> {
  const now = new Date();
  const startOfWeek = startOfISOWeek(now); // or: now - 7 days
  const startOfMonth = new Date(now.getFullYear(), now.getMonth(), 1);
  const thirtyDaysAgo = new Date(now.getTime() - 30 * 86_400_000);
  const twelveWeeksAgo = new Date(now.getTime() - 12 * 7 * 86_400_000);
  const eightWeeksAgo = new Date(now.getTime() - 8 * 7 * 86_400_000);

  const [
    totalUsers,
    newUsersThisWeek,
    newUsersThisMonth,
    totalVehicles,
    totalPosts,
    totalComments,
    aiReportsThisMonth,
    parseStats,
    weeklySignups,
    weeklyActivity,
    pendingReportsCount,
  ] = await Promise.all([
    db.select({ count: sql<number>`COUNT(*)` }).from(users).then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(users)
      .where(gte(users.createdAt, startOfWeek))
      .then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(users)
      .where(gte(users.createdAt, startOfMonth))
      .then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(vehicles).then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(posts).then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(postComments).then(r => Number(r[0]?.count ?? 0)),

    db.select({ count: sql<number>`COUNT(*)` }).from(aiReports)
      .where(and(gte(aiReports.requestedAt, startOfMonth), ne(aiReports.status, "failed")))
      .then(r => Number(r[0]?.count ?? 0)),

    this.getParseSuccessRate(thirtyDaysAgo),

    this.getWeeklySignups(twelveWeeksAgo),

    this.getWeeklyActivity(eightWeeksAgo),

    db.select({ count: sql<number>`COUNT(*)` }).from(postReports)
      .innerJoin(posts, eq(postReports.postId, posts.id))
      .where(eq(posts.isHidden, true))
      .then(r => Number(r[0]?.count ?? 0)),
  ]);

  return {
    totalUsers,
    newUsersThisWeek,
    newUsersThisMonth,
    totalVehicles,
    totalPosts,
    totalComments,
    aiReportsThisMonth,
    parseSuccessRate: parseStats,
    weeklySignups,        // Array<{ week: string; count: number }>
    weeklyActivity,       // Array<{ week: string; posts: number; comments: number }>
    pendingReportsCount,
  };
}

private async getParseSuccessRate(since: Date): Promise<number> {
  const [total, parsed] = await Promise.all([
    db.select({ count: sql<number>`COUNT(*)` }).from(documents)
      .where(gte(documents.createdAt, since))
      .then(r => Number(r[0]?.count ?? 0)),
    db.select({ count: sql<number>`COUNT(*)` }).from(documents)
      .where(and(gte(documents.createdAt, since), eq(documents.parseStatus, "parsed")))
      .then(r => Number(r[0]?.count ?? 0)),
  ]);
  if (total === 0) return 100;
  return Math.round((parsed / total) * 100);
}
```

Weekly series are built using `date_trunc('week', created_at)` SQL aggregation for efficiency.

### Weekly Aggregation Helpers

[Source: architecture.md#6-components]

```typescript
private async getWeeklySignups(since: Date): Promise<Array<{ week: string; count: number }>> {
  const result = await db.execute(sql`
    SELECT
      to_char(date_trunc('week', created_at), 'YYYY-MM-DD') AS week,
      COUNT(*) AS count
    FROM users
    WHERE created_at >= ${since}
    GROUP BY date_trunc('week', created_at)
    ORDER BY week ASC
  `);
  return result.rows.map(r => ({ week: r.week as string, count: Number(r.count) }));
}

private async getWeeklyActivity(since: Date): Promise<Array<{ week: string; posts: number; comments: number }>> {
  // Two queries or a union — use separate queries for clarity
  const [postsByWeek, commentsByWeek] = await Promise.all([
    db.execute(sql`
      SELECT to_char(date_trunc('week', created_at), 'YYYY-MM-DD') AS week, COUNT(*) AS count
      FROM posts WHERE created_at >= ${since}
      GROUP BY week ORDER BY week ASC
    `),
    db.execute(sql`
      SELECT to_char(date_trunc('week', created_at), 'YYYY-MM-DD') AS week, COUNT(*) AS count
      FROM post_comments WHERE created_at >= ${since}
      GROUP BY week ORDER BY week ASC
    `),
  ]);

  // Merge into combined array
  const weeks = new Set([
    ...postsByWeek.rows.map(r => r.week as string),
    ...commentsByWeek.rows.map(r => r.week as string),
  ]);

  return Array.from(weeks).sort().map(week => ({
    week,
    posts: Number(postsByWeek.rows.find(r => r.week === week)?.count ?? 0),
    comments: Number(commentsByWeek.rows.find(r => r.week === week)?.count ?? 0),
  }));
}
```

export async function GET(req: Request) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const analytics = await adminService.getAnalytics();
    return NextResponse.json(analytics);
  } catch (error) {
