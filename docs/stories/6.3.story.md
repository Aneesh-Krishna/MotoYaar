**I want** to edit a trip I've logged,
**so that** I can correct details or add information I missed.

```tsx
// src/app/(app)/trips/[id]/edit/page.tsx (RSC)
import { tripService } from "@/services/tripService";
import { TripEditForm } from "@/components/trips/TripEditForm";

export default async function EditTripPage({ params }: { params: { id: string } }) {
  const session = await getServerSession(authOptions);
  if (!session) redirect("/login");

  try {
    const trip = await tripService.getById(params.id, session.user.id);
    return <TripEditForm trip={trip} />;
  } catch (error) {
    if (error instanceof ForbiddenError) redirect("/trips");
    if (error instanceof NotFoundError) notFound();
    throw error;
  }
}
```

### tripService.update()

[Source: architecture.md#6-components]

Add to `src/services/tripService.ts`:

```typescript
async update(tripId: string, userId: string, data: UpdateTripInput): Promise<Trip> {
  // 1. Ownership check
  const trip = await db.query.trips.findFirst({ where: eq(trips.id, tripId) });
  if (!trip) throw new NotFoundError("Trip not found");
  if (trip.userId !== userId) throw new ForbiddenError("You do not own this trip");

  const newTotalCost = (data.breakdown ?? trip.breakdown)
    .reduce((sum: number, item: TripBreakdownItem) => sum + item.amount, 0);

  // 2. Update trip record
  const [updated] = await db.update(trips)
    .set({
      title: data.title,
      description: data.description,
      startDate: data.startDate ? new Date(data.startDate) : undefined,
      endDate: data.endDate ? new Date(data.endDate) : null,
      vehicleId: data.vehicleId !== undefined ? data.vehicleId ?? null : undefined,
      routeText: data.routeText,
      mapsLink: data.mapsLink,
      timeTaken: data.timeTaken,
      breakdown: data.breakdown ?? trip.breakdown,
    })
    .where(eq(trips.id, tripId))
    .returning();

  // 3. Find auto-created expense linked to this trip
  const linkedExpense = await db.query.expenses.findFirst({
    where: and(eq(expenses.tripId, tripId), eq(expenses.userId, userId)),
  });

  if (linkedExpense) {
    const vehicleChanged = data.vehicleId !== undefined && data.vehicleId !== trip.vehicleId;
    const amountsChanged = data.breakdown !== undefined;

    if (vehicleChanged || amountsChanged) {
      await db.update(expenses)
        .set({
          vehicleId: vehicleChanged ? (data.vehicleId ?? null) : undefined,
          price: amountsChanged ? newTotalCost : undefined,
        })
        .where(eq(expenses.id, linkedExpense.id));
    }
  }

  return updated;
}
```

Key behavior:
- `vehicleId` change: update `expenses.vehicle_id` to new vehicle (or null if vehicle removed)
- `breakdown` change: update `expenses.price` to new sum
- If no linked expense exists: do not create one (edit doesn't auto-create new expenses)
