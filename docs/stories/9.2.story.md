### hotScore Algorithm

[Source: architecture.md#6-components]

Create `src/utils/hotScore.ts`:

```typescript
/**
 * Reddit-style hot score for community posts.
 * Higher score = more trending.
 */
export function hotScore(
  likes: number,
  dislikes: number,
  createdAt: Date
): number {
  const score = likes - dislikes;
  const ageHours = (Date.now() - createdAt.getTime()) / 3_600_000;
  return score / Math.pow(ageHours + 2, 1.5);
}
```

The score is computed server-side in `communityService.listPosts()` after fetching posts with reaction counts. Sort in memory (posts per page are already a manageable set — 20 per page).

For full-feed trending sort, compute score for all non-hidden posts, sort, then paginate. This is acceptable for MVP post counts.

### communityService.listPosts()

[Source: architecture.md#6-components]

Add to `src/services/communityService.ts`:

```typescript
async listPosts(
  sort: "trending" | "newest",
  page: number,
  tag?: string,
  userId?: string
): Promise<{ posts: FeedPost[]; hasMore: boolean }> {
  const limit = 20;

  // Fetch with counts
  const rawPosts = await db.query.posts.findMany({
    where: and(
      eq(posts.isHidden, false),
      tag ? sql`${tag} = ANY(${posts.tags})` : undefined,
    ),
    with: {
      user: { columns: { id: true, name: true, image: true } },
      reactions: { columns: { type: true } },
      comments: { columns: { id: true } },
    },
    // Fetch enough for scoring — for MVP: fetch all and sort (optimise later with pagination)
  });

  // Compute reaction counts and hot score
  const withScores = rawPosts.map(p => {
    const likes = p.reactions.filter(r => r.type === "like").length;
    const dislikes = p.reactions.filter(r => r.type === "dislike").length;
    const commentCount = p.comments.length;
    const score = hotScore(likes, dislikes, p.createdAt);
    const userReaction = userId
      ? p.reactions.find(r => r.userId === userId)?.type ?? null
      : null;
    return { ...p, likes, dislikes, commentCount, score, userReaction };
  });

  // Sort
  const sorted = sort === "trending"
    ? withScores.sort((a, b) => b.score - a.score)
    : withScores.sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime());

  // Pagination
  const offset = (page - 1) * limit;
  const paginated = sorted.slice(offset, offset + limit);

  return { posts: paginated, hasMore: offset + limit < sorted.length };
}
```

Note: For very large post counts this full-fetch approach needs optimisation. For MVP (< 10k posts) it is acceptable. Add a `// TODO: optimise with DB-level hot score computation at scale` comment.

### GET /api/posts

[Source: architecture.md#5-api-specification]

Add `GET` handler to `src/app/api/posts/route.ts` (POST already exists from Story 9.1):

```typescript
export async function GET(req: Request) {
  const session = await getServerSession(authOptions);
  // No auth required for reading feed (guests allowed)

  try {
    const { searchParams } = new URL(req.url);
    const sort = (searchParams.get("sort") as "trending" | "newest") ?? "trending";
    const page = parseInt(searchParams.get("page") ?? "1", 10);
    const tag = searchParams.get("tag") ?? undefined;

    const result = await communityService.listPosts(sort, page, tag, session?.user.id);
    return NextResponse.json(result);
  } catch (error) {
    return handleApiError(error);
  }
}
```

### Infinite Scroll

[Source: front-end-spec.md#4-loading-states--skeletons]

In the community page client component, use `IntersectionObserver` on a sentinel div at the bottom of the list:

```tsx
const sentinelRef = useRef<HTMLDivElement>(null);

useEffect(() => {
  const observer = new IntersectionObserver(
    ([entry]) => {
      if (entry.isIntersecting && hasMore && !isLoadingMore) {
        loadMorePosts();
      }
    },
    { threshold: 0.1 }
  );
  if (sentinelRef.current) observer.observe(sentinelRef.current);
  return () => observer.disconnect();
}, [hasMore, isLoadingMore]);

// At bottom of list:
<div ref={sentinelRef} className="h-4" />
{isLoadingMore && <PostCardSkeleton count={3} />}
```
