### adminService.searchUsers()

[Source: architecture.md#6-components]

Add to `src/services/adminService.ts`:

```typescript
async searchUsers(q: string): Promise<AdminUserSummary[]> {
  const results = await db.query.users.findMany({
    where: q
      ? or(
          ilike(users.email, `%${q}%`),
          ilike(users.name, `%${q}%`),
          ilike(users.username, `%${q}%`),
        )
      : undefined,
    orderBy: [desc(users.createdAt)],
    limit: 50,
  });

  return results;
}

async getUser(userId: string): Promise<AdminUserDetail> {
  const user = await db.query.users.findFirst({
    where: eq(users.id, userId),
  });
  if (!user) throw new NotFoundError("User not found");

  const [vehicleCount, postCount] = await Promise.all([
    db.select({ count: sql<number>`COUNT(*)` }).from(vehicles).where(eq(vehicles.userId, userId))
      .then(r => Number(r[0]?.count ?? 0)),
    db.select({ count: sql<number>`COUNT(*)` }).from(posts).where(eq(posts.userId, userId))
      .then(r => Number(r[0]?.count ?? 0)),
  ]);

  return { ...user, vehicleCount, postCount };
}

async updateUserStatus(
  userId: string,
  action: "warn" | "suspend" | "ban" | "lift" | "unban" | "relink",
  options: { suspendDays?: number; googleId?: string; adminId: string }
): Promise<void> {
  switch (action) {
    case "warn":
      return this.warnUser(userId, options.adminId);
    case "suspend":
      if (!options.suspendDays) throw new Error("suspendDays required");
      return this.suspendUser(userId, options.suspendDays, options.adminId);
    case "ban":
      return this.banUser(userId, options.adminId);
    case "lift":
      await db.update(users)
        .set({ status: "active", suspendedUntil: null })
        .where(eq(users.id, userId));
      logger.info({ action: "lift_suspension", userId, adminId: options.adminId }, "admin action");
      return;
    case "unban":
      await db.update(users)
        .set({ status: "active" })
        .where(eq(users.id, userId));
      logger.info({ action: "unban_user", userId, adminId: options.adminId }, "admin action");
      return;
    case "relink":
      if (!options.googleId) throw new Error("googleId required");
      await db.update(users)
        .set({ googleId: options.googleId })
        .where(eq(users.id, userId));
      logger.info({ action: "relink_google", userId, adminId: options.adminId }, "admin action");
      return;
  }
}
```

### API Routes

[Source: architecture.md#5-api-specification]

Create `src/app/api/admin/users/route.ts`:

```typescript
export async function GET(req: Request) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { searchParams } = new URL(req.url);
  const q = searchParams.get("q") ?? "";

  try {
    const users = await adminService.searchUsers(q);
    return NextResponse.json(users);
  } catch (error) {
    return handleApiError(error);
  }
}
```

Create `src/app/api/admin/users/[id]/route.ts`:

```typescript
export async function GET(req: Request, { params }: { params: { id: string } }) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const user = await adminService.getUser(params.id);
    return NextResponse.json(user);
  } catch (error) {
    return handleApiError(error);
  }
}

export async function PATCH(req: Request, { params }: { params: { id: string } }) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const body = await req.json();
    await adminService.updateUserStatus(params.id, body.action, {
      suspendDays: body.suspendDays,
      googleId: body.googleId,
      adminId: session.admin.id,
    });
    return NextResponse.json({ ok: true });
  } catch (error) {
    return handleApiError(error);
  }
}
```

### Re-link Google Account

[Source: architecture.md#6-components]

The re-link action updates `users.google_id` to a new Google account ID. This allows support to link a user's MotoYaar account to a new Google account if they changed Google accounts. This is a privileged operation — admin only. No notification sent to the user for re-link.

Verify `users.google_id TEXT` column exists in the schema (should be from Story 1.2 auth setup).
