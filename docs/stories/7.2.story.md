// src/app/(app)/reports/page.tsx (RSC)
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { reportService } from "@/services/reportService";
import { redirect } from "next/navigation";

interface PageProps {
  searchParams: {
    filter?: string;    // "monthly" | "range" | "yearly"
    month1?: string;    // YYYY-MM
    month2?: string;    // YYYY-MM
    from?: string;      // YYYY-MM-DD
    to?: string;        // YYYY-MM-DD
    compFrom?: string;  // override comparison range start
    compTo?: string;    // override comparison range end
  };
}

export default async function ReportsPage({ searchParams }: PageProps) {
  const session = await getServerSession(authOptions);
  if (!session) redirect("/login");

  const report = await reportService.getOverallReport(session.user.id, searchParams);
  return <OverallReportView report={report} />;
}
```

`OverallReportView` is a `"use client"` component that manages filter UI state and updates URL query params via `router.replace`.

export async function GET(req: Request) {
  const session = await getServerSession(authOptions);
  if (!session) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const { searchParams } = new URL(req.url);
    const filter = {
      type: searchParams.get("filter") ?? "monthly",
      month1: searchParams.get("month1") ?? undefined,
      month2: searchParams.get("month2") ?? undefined,
      from: searchParams.get("from") ?? undefined,
      to: searchParams.get("to") ?? undefined,
      compFrom: searchParams.get("compFrom") ?? undefined,
      compTo: searchParams.get("compTo") ?? undefined,
    };
    const report = await reportService.getOverallReport(session.user.id, filter);
    return NextResponse.json(report);
  } catch (error) {
    return handleApiError(error);
  }
}
```

### reportService.getOverallReport()

[Source: architecture.md#6-components]

Add to `src/services/reportService.ts`:

```typescript
async getOverallReport(userId: string, filter: OverallReportFilter): Promise<OverallReport> {
  const vehicles = await vehicleService.listByUser(userId);

  // Resolve primary and comparison date ranges
  const { primary, comparison, comparisonLabel } = resolveComparisonRanges(filter);

  // Fetch all expenses for all vehicles in both ranges
  const [primaryExpenses, compExpenses] = await Promise.all([
    expenseService.listByUserAndRange(userId, primary.from, primary.to),
    expenseService.listByUserAndRange(userId, comparison.from, comparison.to),
  ]);

  // Per-vehicle breakdown for primary period
  const perVehicle = vehicles.map(v => ({
    vehicleId: v.id,
    vehicleName: v.name,
    total: sumByVehicle(primaryExpenses, v.id, userId),
  }));

  const totalSpend = perVehicle.reduce((s, v) => s + v.total, 0);
  const prevTotalSpend = sumAll(compExpenses, userId);

  return {
    perVehicle,
    totalSpend,
    prevTotalSpend,
    comparisonLabel,           // e.g. "Nov 2025 vs Oct 2025"
    byCategory: aggregateByCategory(primaryExpenses, userId),
    monthlyData: buildMonthlySeriesComparison(primaryExpenses, compExpenses, userId),
    hadCurrencyConversion: checkHadConversion(primaryExpenses, userId),
  };
}
```

### Comparison Logic

[Source: architecture.md#5-api-specification]

`resolveComparisonRanges(filter)` handles three filter modes:

**Monthly** (`filter.type === "monthly"`):
- Primary: the full calendar month of `month1` (e.g. "2025-11" → Nov 1–Nov 30 2025)
- Comparison: the full calendar month of `month2` (e.g. "2025-10" → Oct 1–Oct 31 2025)
- Label: "Nov 2025 vs Oct 2025"

**Date Range** (`filter.type === "range"`):
- Primary: `from` → `to`
- Auto-comparison: shift back by (to - from + 1 days), so same duration immediately prior
- User override: if `compFrom` and `compTo` provided, use those
- Label: "[from]→[to] vs [compFrom]→[compTo]"

**Yearly** (`filter.type === "yearly"`):
- Primary: Jan 1 this year → today
- Comparison: Jan 1 last year → Dec 31 last year
- Label: "2025 vs 2024"

### Currency Conversion Utility

[Source: architecture.md#section-currency]

Create `src/utils/currency.ts`:

```typescript
// Static exchange rates — MVP, no live API
// Base: INR
const RATES: Record<string, number> = {
  INR: 1,
  USD: 0.012,   // 1 INR = 0.012 USD
  EUR: 0.011,
  GBP: 0.0095,
  AED: 0.044,
  SGD: 0.016,
};

export function convertAmount(
  amount: number,
  fromCurrency: string,
  toCurrency: string
): number {
  if (fromCurrency === toCurrency) return amount;
  const inINR = amount / (RATES[fromCurrency] ?? 1);
  return inINR * (RATES[toCurrency] ?? 1);
}

export function formatCurrency(amount: number, currency: string): string {
  return new Intl.NumberFormat("en-IN", {
    style: "currency",
    currency,
    maximumFractionDigits: 0,
  }).format(amount);
}
```

Note: Exchange rates are hardcoded for MVP. A comment in the file must note: `// TODO: Replace with live rates API in future`. No external API call is made.

### Filter UI

[Source: front-end-spec.md#4-loading-states--skeletons]

Filter mode selector: segmented control with three options — **Monthly** / **Date Range** / **Yearly**. Selecting each reveals different sub-inputs:

- Monthly: two month pickers (month1, month2)
- Date Range: date range picker (from/to) + optional "Override comparison" toggle revealing compFrom/compTo
- Yearly: no sub-inputs (auto-calculates)

All filter changes update URL query params so the page is shareable/bookmarkable.
