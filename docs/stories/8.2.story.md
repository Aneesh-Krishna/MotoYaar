# Story 8.2 — PWA Push Notification Subscription

**Epic:** 08 — Notifications & Background Jobs
**Story:** 8.2
**Status:** Draft

## Story

**As a** user who has completed onboarding,
**I want** to be prompted to enable push notifications,
**so that** I receive timely alerts about document expiry without having to open the app.

---

## Acceptance Criteria

1. - [ ] Push permission prompt shown after onboarding walkthrough completion — NOT on raw page load
2. - [ ] Prompt UI: title "Stay on top of your documents", description "Enable push notifications to get expiry alerts", CTA "Enable" + secondary "Not now"
3. - [ ] On "Enable": call `Notification.requestPermission()` → if granted: call `subscribeToPush()` from `src/lib/push.ts` → `POST /api/push/subscribe` with subscription object
4. - [ ] On "Not now": dismiss prompt; do not ask again on next load (store preference in localStorage: `pushPromptDismissed=true`)
5. - [ ] `POST /api/push/subscribe` saves subscription to `push_subscriptions` table (upsert by endpoint)
6. - [ ] Multiple devices supported: unique endpoint per subscription; same user can have many subscriptions
7. - [ ] Stale subscription handling: when `sendPushNotification` returns 410 (Gone), the subscription is deleted from DB
8. - [ ] User can enable push later from Settings → Notifications (Settings page in Story 8.5)
9. - [ ] `src/hooks/usePushSubscription.ts` — client hook managing permission state and subscription flow
10. - [ ] `src/components/notifications/PushPermissionPrompt.tsx` — client component rendering the prompt
11. - [ ] `pushService.subscribe(userId, subscription)` in `src/services/pushService.ts`
12. - [ ] `pnpm typecheck` and `pnpm lint` pass

### DB Schema: push_subscriptions Table

[Source: architecture.md#3-database-schema]

```sql
CREATE TABLE push_subscriptions (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  endpoint    TEXT NOT NULL UNIQUE,
  p256dh      TEXT NOT NULL,
  auth        TEXT NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX idx_push_subscriptions_user ON push_subscriptions(user_id);
```

The `endpoint` is the browser push service URL (unique per browser/device registration). The `p256dh` and `auth` fields come from the `PushSubscription` object.

### usePushSubscription Hook

[Source: architecture.md#6-components]

Create `src/hooks/usePushSubscription.ts`:

```typescript
"use client";
import { useState, useEffect } from "react";
import { subscribeToPush } from "@/lib/push";

export type PushPermissionState = "default" | "granted" | "denied" | "unsupported";

export function usePushSubscription() {
  const [permissionState, setPermissionState] = useState<PushPermissionState>("default");
  const [isSubscribing, setIsSubscribing] = useState(false);

  useEffect(() => {
    if (typeof Notification === "undefined") {
      setPermissionState("unsupported");
      return;
    }
    setPermissionState(Notification.permission as PushPermissionState);
  }, []);

  const subscribe = async (): Promise<boolean> => {
    if (typeof Notification === "undefined") return false;
    setIsSubscribing(true);
    try {
      const permission = await Notification.requestPermission();
      setPermissionState(permission as PushPermissionState);
      if (permission !== "granted") return false;

      // subscribeToPush() from src/lib/push.ts (client-side)
      const subscription = await subscribeToPush();
      if (!subscription) return false;

      // Send to server
      await fetch("/api/push/subscribe", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(subscription.toJSON()),
      });
      return true;
    } catch (err) {
      console.error("Push subscription failed", err);
      return false;
    } finally {
      setIsSubscribing(false);
    }
  };

  const isDismissed = (): boolean => {
    if (typeof window === "undefined") return false;
    return localStorage.getItem("pushPromptDismissed") === "true";
  };

  const dismiss = () => {
    localStorage.setItem("pushPromptDismissed", "true");
  };

  return { permissionState, subscribe, dismiss, isDismissed, isSubscribing };
}
```

"use client";

### POST /api/push/subscribe

[Source: architecture.md#5-api-specification]

Create `src/app/api/push/subscribe/route.ts`:

```typescript
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { handleApiError } from "@/lib/errors";
import { pushService } from "@/services/pushService";
import { NextResponse } from "next/server";

export async function POST(req: Request) {
  const session = await getServerSession(authOptions);
  if (!session) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const body = await req.json();
    // body: { endpoint, keys: { p256dh, auth } }
    await pushService.subscribe(session.user.id, {
      endpoint: body.endpoint,
      p256dh: body.keys.p256dh,
      auth: body.keys.auth,
    });
    return NextResponse.json({ ok: true });
  } catch (error) {
    return handleApiError(error);
  }
}
```

### pushService.subscribe()

[Source: architecture.md#6-components]

Create `src/services/pushService.ts`:

```typescript
import { db } from "@/db";
import { pushSubscriptions } from "@/db/schema";
import { eq } from "drizzle-orm";

interface SubscriptionData {
  endpoint: string;
  p256dh: string;
  auth: string;
}

export const pushService = {
  async subscribe(userId: string, data: SubscriptionData): Promise<void> {
    // Upsert by endpoint (unique constraint)
    await db.insert(pushSubscriptions)
      .values({ userId, ...data })
      .onConflictDoUpdate({
        target: pushSubscriptions.endpoint,
        set: { userId, p256dh: data.p256dh, auth: data.auth },
      });
  },

  async getSubscriptionsForUser(userId: string) {
    return db.query.pushSubscriptions.findMany({
      where: eq(pushSubscriptions.userId, userId),
    });
  },

  async deleteSubscription(endpoint: string): Promise<void> {
    await db.delete(pushSubscriptions)
      .where(eq(pushSubscriptions.endpoint, endpoint));
  },
};
```

### subscribeToPush() in src/lib/push.ts

[Source: architecture.md#11-integrations]

Read `src/lib/push.ts` before calling `subscribeToPush()`. The function signature from architecture:
- `subscribeToPush()` (client-side): registers the service worker push subscription and returns a `PushSubscription` object
- It uses `NEXT_PUBLIC_VAPID_PUBLIC_KEY` environment variable

If the implementation is only a stub, the agent must implement it:

```typescript
// src/lib/push.ts (client-side portion)
export async function subscribeToPush(): Promise<PushSubscription | null> {
  if (!("serviceWorker" in navigator) || !("PushManager" in window)) return null;

  const registration = await navigator.serviceWorker.ready;
  const subscription = await registration.pushManager.subscribe({
    userVisibleOnly: true,
    applicationServerKey: urlBase64ToUint8Array(
      process.env.NEXT_PUBLIC_VAPID_PUBLIC_KEY!
    ),
  });
  return subscription;
}

function urlBase64ToUint8Array(base64String: string): Uint8Array {
  const padding = "=".repeat((4 - (base64String.length % 4)) % 4);
  const base64 = (base64String + padding).replace(/-/g, "+").replace(/_/g, "/");
  const rawData = window.atob(base64);
  return Uint8Array.from([...rawData].map(c => c.charCodeAt(0)));
}
```
