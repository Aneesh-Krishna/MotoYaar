### Route and Page Structure

```tsx
// src/app/(app)/garage/[id]/page.tsx (RSC)
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { vehicleService } from "@/services/vehicleService";
import { redirect, notFound } from "next/navigation";
import { ForbiddenError, NotFoundError } from "@/lib/errors";

interface Props {
  params: { id: string };
  searchParams: { tab?: string };
}

export default async function VehicleDetailPage({ params, searchParams }: Props) {
  const session = await getServerSession(authOptions);
  if (!session) redirect("/login");

  try {
    const vehicle = await vehicleService.getWithAccessCheck(params.id, session.user.id);
    const activeTab = searchParams.tab ?? "overview";

    return (
      <div>
        <VehicleHeader vehicle={vehicle} />
        <VehicleDetailTabs vehicle={vehicle} activeTab={activeTab} />
      </div>
    );
  } catch (error) {
    if (error instanceof ForbiddenError) redirect("/garage");
    if (error instanceof NotFoundError) notFound();
    throw error;
  }
}
```

### vehicleService.getWithAccessCheck()

[Source: architecture.md#6-components]

Add to `src/services/vehicleService.ts`:

```typescript
async getWithAccessCheck(vehicleId: string, userId: string): Promise<Vehicle> {
  const vehicle = await db.query.vehicles.findFirst({
    where: eq(vehicles.id, vehicleId),
  });

  if (!vehicle) throw new NotFoundError("Vehicle not found");

  // Owner check
  if (vehicle.userId === userId) return vehicle;

  // Viewer check (VehicleAccess table)
  const access = await db.query.vehicleAccess.findFirst({
    where: and(
      eq(vehicleAccess.vehicleId, vehicleId),
      eq(vehicleAccess.viewerId, userId)
    ),
  });

  if (!access) throw new ForbiddenError("You do not have access to this vehicle");

  return vehicle;
}
```

The `vehicle_access` table (from DB schema) stores viewer relationships. Check it here before throwing 403.

### Overview Tab Content

[Source: epic-03-vehicle-management.md#STORY-03-04]

The Overview tab renders three metric cards:

1. **Total Spend**: `vehicleService.getTotalSpend(vehicleId)` — SUM of all expenses. Format as `₹1,23,456` using Indian number formatting.
2. **Last Service Date**: Most recent expense where `reason = "Service"`. Format: `15 Mar 2026`. Show "No service recorded" if none.
3. **Next Doc Expiry**: Nearest upcoming `expiryDate` from documents for this vehicle. Show StatusBadge next to the date. Show "No documents added" placeholder if no docs.

These can be computed server-side in the page component and passed as props.

```typescript
]);
```

### DB Schema Reference

[Source: architecture.md#9-database-schema]

The `vehicles` table has `user_id` FK. The `vehicle_access` table (for shared vehicles) uses:
```sql
CREATE TABLE vehicle_access (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  vehicle_id  UUID NOT NULL REFERENCES vehicles(id) ON DELETE CASCADE,
  viewer_id   UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  invited_by  UUID NOT NULL REFERENCES users(id),
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(vehicle_id, viewer_id)
);
```

In `vehicleService.getWithAccessCheck`, query both `vehicles` and `vehicle_access`.
