# Story 8.1 — Daily Cron Job for Document Expiry Checking

## Story

**As a** vehicle owner,
**I want** to be automatically notified when my documents are nearing expiry or have expired,
**so that** I never get caught out by an expired licence, insurance, or registration.

import { cronService } from "@/services/cronService";
import { logger } from "@/lib/logger";

export async function GET(req: Request) {
  const authHeader = req.headers.get("Authorization");
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  try {
    const result = await cronService.runDocumentExpiryCheck();
    logger.info({ ...result }, "Document expiry cron completed");
    return NextResponse.json(result);
  } catch (err) {
    logger.error({ err }, "Document expiry cron failed");
    return NextResponse.json({ error: "Cron failed" }, { status: 500 });
  }
}

### cronService.runDocumentExpiryCheck()

[Source: architecture.md#6-components]

Create `src/services/cronService.ts`:

```typescript
import { db } from "@/db";
import { documents, users } from "@/db/schema";
import { isNotNull, lt, lte, isNull, and } from "drizzle-orm";
import { notificationService } from "./notificationService";
import { emailService } from "./emailService";
import { pushService } from "./pushService";
import { deleteObject } from "@/lib/r2";
import { logger } from "@/lib/logger";

const MS_PER_DAY = 86_400_000;

export const cronService = {
  async runDocumentExpiryCheck(): Promise<{ processed: number; notified: number; deleted: number }> {
    const now = new Date();
    const today = new Date(now.getFullYear(), now.getMonth(), now.getDate());

    // Step 1: Fetch all documents with an expiry date
    const allDocs = await db.query.documents.findMany({
      where: isNotNull(documents.expiryDate),
      with: { user: true },
    });

    let notified = 0;
    let deleted = 0;

    // Group by userId for batching
    const byUser = new Map<string, { user: User; warningDocs: Doc[]; expiredDocs: Doc[] }>();

    for (const doc of allDocs) {
      try {
        const expiryDate = new Date(doc.expiryDate!);
        const daysUntilExpiry = Math.ceil((expiryDate.getTime() - today.getTime()) / MS_PER_DAY);
        const windowDays = doc.user.notificationWindowDays ?? 30;

        // Step 5: Mark as expired
        if (daysUntilExpiry < 0) {
          await db.update(documents)
            .set({ status: "expired" })
            .where(eq(documents.id, doc.id));
        }

        // Step 6: Delete R2 file if expired > 10 days ago
        if (daysUntilExpiry < -10 && doc.storageUrl) {
          try {
            await deleteObject(doc.storageUrl);
            await db.update(documents)
              .set({ storageUrl: null })
              .where(eq(documents.id, doc.id));
            deleted++;
          } catch (err) {
            logger.warn({ err, docId: doc.id }, "R2 delete failed");
          }
        }

        // Step 3: Warning notification
        if (daysUntilExpiry > 0 && daysUntilExpiry <= windowDays && !doc.expiryWarningNotifiedAt) {
          const entry = getOrCreate(byUser, doc.user.id, doc.user);
          entry.warningDocs.push(doc);
          await db.update(documents)
            .set({ expiryWarningNotifiedAt: now })
            .where(eq(documents.id, doc.id));
        }

        // Step 4: Expiry-day notification
        if (daysUntilExpiry === 0 && !doc.expiryNotifiedAt) {
          const entry = getOrCreate(byUser, doc.user.id, doc.user);
          entry.expiredDocs.push(doc);
          await db.update(documents)
            .set({ expiryNotifiedAt: now })
            .where(eq(documents.id, doc.id));
        }
      } catch (err) {
        logger.error({ err, docId: doc.id }, "Error processing document in cron");
        // Continue to next document
      }
    }

    // Send batched notifications per user
    for (const [userId, { user, warningDocs, expiredDocs }] of byUser) {
      try {
        const allNotifyDocs = [...warningDocs, ...expiredDocs];
        if (allNotifyDocs.length > 0) {
          await sendBatchedDocumentNotification(userId, user, warningDocs, expiredDocs);
          notified++;
        }
      } catch (err) {
        logger.error({ err, userId }, "Error sending batch notification");
      }
    }

    return { processed: allDocs.length, notified, deleted };
  }
};
```

### Batched Notification Logic

[Source: architecture.md#11-integrations]

`sendBatchedDocumentNotification()` is a private helper that creates ONE notification per user regardless of how many documents are expiring:

```typescript
async function sendBatchedDocumentNotification(
  userId: string,
  user: User,
  warningDocs: Doc[],
  expiredDocs: Doc[]
): Promise<void> {
  const allDocs = [...warningDocs, ...expiredDocs];
  const docNames = allDocs.map(d => d.type).join(", ");

  // In-app notification (ONE per user per run)
  const hasExpired = expiredDocs.length > 0;
  const title = hasExpired
    ? `Document${expiredDocs.length > 1 ? "s" : ""} expired`
    : `Document expiry reminder`;
  const body = hasExpired
    ? `Your ${docNames} has expired. Please renew.`
    : `Your ${docNames} is expiring soon.`;

  await notificationService.create(userId, "document_expiry", title, body, "/garage");

  // Email
  await emailService.sendDocumentExpiryEmail(userId, allDocs);

  // Push (if user has enabled it)
  if (user.pushNotificationsEnabled) {
    const subscriptions = await pushService.getSubscriptionsForUser(userId);
    await Promise.allSettled(
      subscriptions.map(sub =>
        sendPushNotification(sub, { title, body, url: "/garage" })
          .catch(async err => {
            if (err.statusCode === 410) await pushService.deleteSubscription(sub.endpoint);
          })
      )
    );
  }
}
```
