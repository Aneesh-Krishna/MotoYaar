```tsx
import { getServerSession } from "next-auth";

import { tripService } from "@/services/tripService";

export default async function TripsPage() {

      <h1 className="text-xl font-bold text-gray-900 mb-4">Trips</h1>
      <TripsListView trips={trips} vehicles={vehicles} />
    </div>
  );
}
```

`TripsListView` is a `"use client"` component that manages the vehicle filter state.

### tripService.listByUser()

[Source: architecture.md#6-components]

Add to `src/services/tripService.ts`:

```typescript
async listByUser(userId: string): Promise<Trip[]> {
  return db.query.trips.findMany({
    where: eq(trips.userId, userId),
    orderBy: [desc(trips.startDate)],
    with: {
      vehicle: {
        columns: { id: true, name: true },
      },
    },
  });
}
```

The `with: { vehicle }` join uses Drizzle's relational query API to include the linked vehicle name.

Card layout:
```
┌─────────────────────────────────────────────────────┐
│  [Title]                              [Total Cost]  │
│  [Date range]    [Vehicle name or "No vehicle"]     │
└─────────────────────────────────────────────────────┘
```

```tsx
<Link href={`/trips/${trip.id}`}>
  <div className="bg-white rounded-xl border border-gray-200 px-4 py-4 flex items-center justify-between active:bg-gray-50">
    <div className="flex-1 min-w-0 pr-4">
      <h3 className="font-semibold text-gray-900 text-sm truncate">{trip.title}</h3>
      <div className="flex items-center gap-2 mt-0.5 text-xs text-gray-500">
        <span>{formatTripDateRange(trip.startDate, trip.endDate)}</span>
        {trip.vehicle ? (
          <>
            <span className="text-gray-300">·</span>
            <span className="flex items-center gap-1">
              <Car size={10} />
              {trip.vehicle.name}
            </span>
          </>
        ) : (
          <>
            <span className="text-gray-300">·</span>
            <span className="text-gray-400">No vehicle</span>
          </>
        )}
      </div>
    </div>
    <div className="text-right flex-shrink-0">
      <p className="font-bold text-gray-900 text-sm">
        {tripTotal > 0 ? formatINR(tripTotal) : "—"}
      </p>
    </div>
  </div>
</Link>
```

Compute `tripTotal` from `trip.breakdown.reduce((s, i) => s + i.amount, 0)`.
