### communityService.getPost()

[Source: architecture.md#6-components]

```typescript
async getPost(postId: string, userId?: string): Promise<PostDetail> {
  const post = await db.query.posts.findFirst({
    where: eq(posts.id, postId),
    with: {
      user: { columns: { id: true, name: true, image: true } },
      reactions: true,
      comments: {
        with: {
          user: { columns: { id: true, name: true, image: true } },
          replies: {
            with: {
              user: { columns: { id: true, name: true, image: true } },
            },
            orderBy: [asc(postComments.createdAt)],
          },
        },
        where: isNull(postComments.parentCommentId), // top-level only
        orderBy: [asc(postComments.createdAt)],
      },
    },
  });

  if (!post) throw new NotFoundError("Post not found");

  const likes = post.reactions.filter(r => r.type === "like").length;
  const dislikes = post.reactions.filter(r => r.type === "dislike").length;
  const userReaction = userId
    ? post.reactions.find(r => r.userId === userId)?.type ?? null
    : null;

  return { ...post, likes, dislikes, userReaction };
}
```

`communityService.addReaction()`:

```typescript
async addReaction(postId: string, userId: string, type: "like" | "dislike"): Promise<{ likes: number; dislikes: number; userReaction: string | null }> {
  // Check existing reaction
  const existing = await db.query.postReactions.findFirst({
    where: and(eq(postReactions.postId, postId), eq(postReactions.userId, userId)),
  });

  if (existing?.type === type) {
    // Same reaction → unlike (remove)
    await db.delete(postReactions)
      .where(and(eq(postReactions.postId, postId), eq(postReactions.userId, userId)));
  } else {
    // Different or no reaction → upsert
    await db.insert(postReactions)
      .values({ postId, userId, type })
      .onConflictDoUpdate({
        target: [postReactions.postId, postReactions.userId],
        set: { type },
      });
  }

  // Return updated counts
  const reactions = await db.query.postReactions.findMany({
    where: eq(postReactions.postId, postId),
  });
  const likes = reactions.filter(r => r.type === "like").length;
  const dislikes = reactions.filter(r => r.type === "dislike").length;
  const userReaction = reactions.find(r => r.userId === userId)?.type ?? null;

  return { likes, dislikes, userReaction };
}
```

### Image Carousel

[Source: front-end-spec.md#5-component-library--ui-patterns]

For posts with 2 images, render a swipeable carousel. Use Framer Motion `motion.div` with drag constraints:

```tsx
function ImageCarousel({ imageUrls }: { imageUrls: string[] }) {
  const [current, setCurrent] = useState(0);

  if (imageUrls.length === 1) {
    return <img src={generateAccessUrl(imageUrls[0])} className="w-full rounded-xl" />;
  }

  return (
    <div className="relative overflow-hidden rounded-xl">
      <motion.div
        drag="x"
        dragConstraints={{ left: 0, right: 0 }}
        onDragEnd={(_, info) => {
          if (info.offset.x < -50 && current < imageUrls.length - 1) setCurrent(c => c + 1);
          if (info.offset.x > 50 && current > 0) setCurrent(c => c - 1);
        }}
        animate={{ x: `-${current * 100}%` }}
        className="flex"
      >
        {imageUrls.map(key => (
          <img key={key} src={generateAccessUrl(key)} className="w-full flex-shrink-0" />
        ))}
      </motion.div>
      {/* Dot indicators */}
      <div className="absolute bottom-2 left-1/2 -translate-x-1/2 flex gap-1">
        {imageUrls.map((_, i) => (
          <span key={i} className={`w-1.5 h-1.5 rounded-full ${i === current ? "bg-white" : "bg-white/50"}`} />
        ))}
      </div>
    </div>
  );
}
```

```typescript
export async function GET(req: Request, { params }: { params: { id: string } }) {
  // No auth required
  const comments = await communityService.getComments(params.id);
  return NextResponse.json(comments);
}

export async function POST(req: Request, { params }: { params: { id: string } }) {
  const session = await getServerSession(authOptions);
  if (!session) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const { content, parentCommentId } = await req.json();
    const comment = await communityService.addComment(params.id, session.user.id, content, parentCommentId);
    return NextResponse.json(comment, { status: 201 });
  } catch (error) {
    return handleApiError(error);
  }
}
```
