## Story

**As an** admin,
**I want** to create official posts attributed to MotoYaar and pin important content to the top of the feed,
**so that** I can seed the community with quality content and surface important announcements.

// Fetch regular posts (excluding pinned ones)

Pinned posts are always shown first, regardless of trending score or sort order. They do NOT count against the page limit of regular posts.

### POST /api/admin/posts

[Source: architecture.md#5-api-specification]

Create `src/app/api/admin/posts/route.ts`:

```typescript
export async function POST(req: Request) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const body = await req.json();
    const data = createPostSchema.parse(body);

    // Get or look up admin system user ID
    const adminUser = await userService.getByEmail("official@motoyaar.app");
    if (!adminUser) throw new Error("Admin system user not seeded");

    const post = await communityService.createPost(adminUser.id, {
      ...data,
      isPinned: body.isPinned ?? false,
    });

    if (body.isPinned) {
      await db.update(posts)
        .set({ isPinned: true, pinnedAt: new Date() })
        .where(eq(posts.id, post.id));
    }

    return NextResponse.json(post, { status: 201 });
  } catch (error) {
    return handleApiError(error);
  }
}
```

### PATCH /api/admin/posts/[id]/pin

[Source: architecture.md#5-api-specification]

Create `src/app/api/admin/posts/[id]/pin/route.ts`:

```typescript
export async function PATCH(req: Request, { params }: { params: { id: string } }) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const { isPinned } = await req.json();
    await db.update(posts)
      .set({
        isPinned: !!isPinned,
        pinnedAt: isPinned ? new Date() : null,
      })
      .where(eq(posts.id, params.id));
    return NextResponse.json({ ok: true });
  } catch (error) {
    return handleApiError(error);
  }
}
```
