```typescript
import { z } from "zod";

const inviteSchema = z.object({
  emails: z.array(z.string()).min(1).max(100),
});

export async function POST(req: Request) {
  const session = await getAdminSession();
  if (!session.admin) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  try {
    const { emails } = inviteSchema.parse(await req.json());
    const result = await adminService.bulkInviteUsers(emails, session.admin.id);
    return NextResponse.json(result);
  } catch (error) {
    return handleApiError(error);
  }
}
```

### adminService.bulkInviteUsers()

[Source: architecture.md#6-components]

Add to `src/services/adminService.ts`:

```typescript
async bulkInviteUsers(
  rawEmails: string[],
  adminId: string
): Promise<{ sent: number; alreadyRegistered: number; invalid: number; failed: number }> {
  const results = { sent: 0, alreadyRegistered: 0, invalid: 0, failed: 0 };

  // Parse and validate emails
  const validEmails = rawEmails
    .map(e => e.trim().toLowerCase())
    .filter(e => {
      if (!e) return false;
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(e)) { results.invalid++; return false; }
      return true;
    });

  for (const email of validEmails) {
    // Check if already registered
    const existingUser = await db.query.users.findFirst({
      where: eq(users.email, email),
    });

    if (existingUser) {
      results.alreadyRegistered++;
      continue;
    }

    // Send invite
    try {
      await emailService.sendBetaInviteEmail(email, "https://motoyaar.app/login");

      // Track in admin_invites (upsert — allow re-invite)
      await db.insert(adminInvites)
        .values({ email })
        .onConflictDoUpdate({ target: adminInvites.email, set: { sentAt: new Date() } });

      results.sent++;
    } catch (err) {
      logger.warn({ err, email }, "Failed to send beta invite");
      results.failed++;
    }
  }

  logger.info({ ...results, adminId }, "Bulk invite completed");
  return results;
}
```
