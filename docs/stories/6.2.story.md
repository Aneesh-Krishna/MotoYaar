```tsx
// src/app/(app)/trips/[id]/page.tsx (RSC)
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { tripService } from "@/services/tripService";
import { expenseService } from "@/services/expenseService";
import { redirect, notFound } from "next/navigation";
import { ForbiddenError, NotFoundError } from "@/lib/errors";

interface Props {
  params: { id: string };
}

export default async function TripDetailPage({ params }: Props) {
  const session = await getServerSession(authOptions);
  if (!session) redirect("/login");

  try {
    const [trip, linkedExpense] = await Promise.all([
      tripService.getById(params.id, session.user.id),
      expenseService.getByTripId(params.id, session.user.id),
    ]);

    return (
      <div className="px-4 py-4 space-y-6">
        <TripDetailHeader trip={trip} />
        <TripDetailContent trip={trip} linkedExpense={linkedExpense} />
      </div>
    );
  } catch (error) {
    if (error instanceof ForbiddenError) redirect("/trips");
    if (error instanceof NotFoundError) notFound();
    throw error;
  }
}
```

```typescript
async getById(tripId: string, userId: string): Promise<Trip> {
  const trip = await db.query.trips.findFirst({
    where: eq(trips.id, tripId),
    with: {
      vehicle: {
        columns: { id: true, name: true, registrationNumber: true },
      },
    },
  });

  if (!trip) throw new NotFoundError("Trip not found");
  if (trip.userId !== userId) throw new ForbiddenError("You do not have access to this trip");

  return trip;
}
```

```typescript
async getByTripId(tripId: string, userId: string): Promise<Expense | null> {
  const expense = await db.query.expenses.findFirst({
    where: and(
      eq(expenses.tripId, tripId),
      eq(expenses.userId, userId)
    ),
  });
  return expense ?? null;
}
```

### Date Range Formatting

```typescript
// src/utils/date.ts (implement the stub from Story 1.1)
import { format, isSameDay } from "date-fns";

export function formatTripDateRange(startDate: string, endDate?: string | null): string {
  const start = new Date(startDate);
  if (!endDate || isSameDay(start, new Date(endDate))) {
    return format(start, "d MMM yyyy");
  }
  const end = new Date(endDate);
  if (start.getMonth() === end.getMonth() && start.getFullYear() === end.getFullYear()) {
    return `${format(start, "d")}–${format(end, "d MMM yyyy")}`;
  }
  return `${format(start, "d MMM")}–${format(end, "d MMM yyyy")}`;
}
```
